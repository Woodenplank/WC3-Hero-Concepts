Guard loop
    Events
        Time - Every 0.50 seconds of game time
    Conditions
    Actions
        For each (Integer Guard_MUI3) from 1 to Guard_MUI1, do (Actions)
            Loop - Actions
                -------- convenience --------
                Set VariableSet Guard_ti = Guard_MUI2[Guard_MUI3]
                Set VariableSet Guard_dur[Guard_ti] = (Guard_dur[Guard_ti] - 0.50)
                If (All Conditions are True) then do (Then Actions) else do (Else Actions)
                    If - Conditions
                        Guard_dur[Guard_ti] Less than or equal to 0.00
                    Then - Actions
                        Special Effect - Destroy Guard_attach[Guard_ti]
                        Unit Group - Remove Guard_caster[Guard_ti] from Guard_ug.
                        Animation - Remove the Defend animation tag to Guard_caster[Guard_ti]
                        Animation - Add the Lumber animation tag to Guard_caster[Guard_ti]
                        -------- Recycle indices --------
                        Set VariableSet Guard_MUI2[Guard_MUI3] = Guard_MUI2[Guard_MUI1]
                        Set VariableSet Guard_MUI2[Guard_MUI1] = Guard_ti
                        Set VariableSet Guard_MUI1 = (Guard_MUI1 - 1)
                        Set VariableSet Guard_MUI3 = (Guard_MUI3 - 1)
                        If (All Conditions are True) then do (Then Actions) else do (Else Actions)
                            If - Conditions
                                Guard_MUI1 Equal to 0
                            Then - Actions
                                Trigger - Turn off (This trigger)
                                Trigger - Turn off Guard deflect <gen>
                            Else - Actions
                    Else - Actions
